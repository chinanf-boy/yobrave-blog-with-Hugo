# 继承

[复制下面代码 - 点击https://babeljs.io/repl/](https://babeljs.io/repl/)

``` javascript
class b {
  constructor(c,d){
    c = c+'1111';
    console.log(c,d);
  }
  start(){
    console.log('ba ba')
  }
}

class a extends b {
  constructor(c,b){
    super(c,b);
    c = c+'222222';
    console.log(1,c,2,b);
    return this.start;
  } 
  start(){
    console.log('son');
  }
}

var o1 = new a('first', 'second');

console.log('------------------');
console.log(o1.toString());
```
